    rand.Seed(time.Now().UnixNano())

    max := siteCf.AvatarMaxId // 2558
    sampleID := rand.Intn(max-min+1) + min

    uobj.Avatar = uidStr
    uobj.IgnoreLimitedUsers = true

    jb, _ := json.Marshal(uobj)
